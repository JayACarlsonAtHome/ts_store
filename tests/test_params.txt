# ts_store test parameters
#
# DISK_TYPE: x7k (7200rpm), 10k (10000rpm), ssd -- all exactly 3 chars
#   so that disk subdirs line up vertically when ls'ing: x7k/ 10k/ ssd/
#   (CLI --disk accepts 7k/7200/x7k etc. and normalizes)
#
# SIZE: smoke (tiny, ~100 events, safe) or full (heavy stress).
#
# In SIZE=full the runner uses *progressive* difficulty so that:
#   001     : hello-world sanity (a few hundred events total)
#   002     : small (~few thousand)
#   003-004 : medium
#   005/006/007 : heavy tests (100 threads × 10000 events = 1M events per run × 5 runs)
#   006     : heavy writer stress (medium-heavy)
#
# This prevents "test 001 taking an hour" when you want full stress on the big tests.
# The old blunt global override has been replaced by per-test scaling in the runner.
#
# OS_ID is normally **auto-detected** by the runner (you do not set it).
# The code runs uname + parses /etc/os-release to identify the OS,
# then looks up (or creates) the next OS_001 / OS_002 / OS_003... entry in
# the visible test-results/OS_MAP.txt.
#
# You can set OS_ID=OS_001 here (or use --os-id on the command line) as an
# **override** for special cases / testing (debugging detection, weird
# containers/WSL/CI, result migration, etc.). We keep the override because
# it's useful in a real multi-machine testing framework.
#
# The OS_MAP.txt will contain the human list:
#   OS_001 = RHEL 9.6
#   OS_002 = RHEL 10.2
#   OS_003 = Linux Mint
#
# Result directories stay short for alignment (padded OS_00n):
#   test-results/OS_001/x7k/xFull/...
#   test-results/OS_002/10k/Smoke/...
#
# Inside every leaf directory you will also find OS_INFO.txt with the
# full details of the OS + run.
#
# SIZE_LABEL is auto-derived:
#   SIZE=smoke  -> Smoke
#   SIZE=full   -> xFull
#
# If no OS_ID (auto or override), the older flat layout test-results/<DISK_TYPE>/ is used.
DISK_TYPE=ssd
SIZE=smoke
